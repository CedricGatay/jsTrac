 21  /**
 22   * Convert a SVG to PNG.
 23   * Utility method used in jsAnnotate but can also be useful for preprocessing.
 24   * @see http://code.google.com/p/canvg/
 25   * @param svg the svg to convert. See canvg for details on format.
 26   * @param width width of the svg
 27   * @param height height of the svg
 28   */

 30  function svg2png(svg,width,height){
 31     $('body').append("<canvas id='svgCanvas' width='"+width+"' height='"+height+"' ></canvas>");
 32     var canvas = document.getElementById('svgCanvas');
 33     canvg(canvas,svg);
 34 	var ctx = document.getElementById('svgCanvas').getContext('2d');
 35 	var img=ctx.getImageData(0,0,ctx.canvas.width,ctx.canvas.height);
 36     var base64 = canvas.toDataURL('image/png');
 37     $('#svgCanvas').remove();
 38     return base64;
 39  }

 41 /**
 42  * Based on http://jsfiddle.net/rRtAq/2039/ and  http://taitems.tumblr.com/post/549973287/drawing-arrows-in-raphaeljs
 43  * Mixed the 2 to create an arrow
 44  * This function draw arrows by dragging
 45  * @class Line
 46  * @param startX X coordinate of starting point
 47  * @param startY Y coordinate of starting point
 48  * @param endX X coordinate of the arrow end
 49  * @param endY Y coordinate of the arrow end
 50  * @param raphael the raphael canvas
 51  */

 55 function Line(startX, startY, endX, endY, raphael) {
 56     var start = {
 57         x: startX,
 58         y: startY
 59     };
 60     var end = {
 61         x: endX,
 62         y: endY
 63     };
 64     var getPath = function() {
 65 
 66         return "M" + start.x + " " + start.y + " L" + end.x + " " + end.y  ;
 67     };
 68 	//Using the provided arrow end of raphaeljs doesn't work with svg2png, so we draw a path for the arrow
 69     var getArrowPath = function() {
 70             size=10;
 71             return "M "+ end.x + " " + end.y + " L " + (end.x - size) + " " + (end.y - size) +  " L " + (end.x - size) + " " + (end.y + size) + " L " +     end.x + " " + end.y ;
 72         };
 73     var redraw = function() {
 74         var angle = Math.atan2(start.x-end.x,end.y-start.y);
 75         angle = (angle / (2 * Math.PI)) * 360;
 76 
 77         var angle = Raphael.angle(start.x,start.y,end.x,end.y);
 78         arrow.attr("path",getArrowPath());
 79         arrow.transform("r"+(angle+180)+","+end.x+","+end.y);
 80         node.attr("path", getPath());
 81     }
 82 
 83 
 84     var node = raphael.path(getPath());
 85     var arrow = raphael.path(getArrowPath());
 86 
 87 
 88     arrow.attr("stroke","#ff0000");
 89     arrow.attr("fill","#ff0000");
 90     //We use jQuery's attr method for stroke-width, Raphael's attr generates errors in the console log.
 91     $('#arrowOverlay').find('path').attr("stroke-width","7");
 92     node.attr("stroke","#ff0000");
 93     return {
 94         updateStart: function(x, y) {
 95             start.x = x;
 96             start.y = y;
 97             redraw();
 98             return this;
 99         },
100         updateEnd: function(x, y) {
101             end.x = x;
102             end.y = y;
103             redraw();
104             return this;
105         },
106         getNode: function(){
107             return node;
108         },
109         getArrow: function(){
110             return arrow;
111          }
112     };
113 }
