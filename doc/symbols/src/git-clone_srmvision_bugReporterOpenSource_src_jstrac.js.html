 25 /**
 26 * RPC method
 27 * @param Remote RPC Method to call
 28 * @param Parameter additional parameters to the RPC method, not compulsory
 29 * @return Data Formated by the RPC method after XML parsing
 30 * @throw Exception Either an error object generated from connection error or error from the Trac
 31 * */
 32 
 33 function rpcTrac(method) {
 34 	var request = new XmlRpcRequest(option.tracUrl+option.rpcPlugin, method);
 35 
 36 	for (var i=1;i<arguments.length;i++){
 37 		request.addParam(arguments[i]);
 38 	}
 39 	try{
 40 		var data = request.send().parseXML();
 41 	}
 42 	catch(err){
 43 		throw err;
 44 	}
 45 	if (typeof data.faultCode!='undefined'){
 46 		throw data;
 47 	}
 48 	return data;
 49 };

 53 /**
 54 * Method creating a select dropdown list in the body
 55 * @param Data array provided by the RPC method
 56 * @param Name Name/id of the select tag
 57 * @param Title Label to display
 58 * @parem Append Where it will be appended
 59 * @param defaultValue Select a value in the select if it exists
 60 */
 61 
 62 function createSelect(data, name, title, append, defaultValue) {
 63 	createDiv=$('<div>').addClass('tracSelect')
 64 	.append("<label for='" + title + "'>" + title + "</label>").appendTo(append);
 65 	selectTag=$('<select>').attr('id',name).attr('name',name).appendTo(createDiv);
 66 	var defaultFound=false;
 67 	$.each(data, function(index, value) {
 68 		if (defaultValue!= null && value.toLowerCase()==defaultValue.toLowerCase()){
 69 			defaultFound=true;
 70 			selectTag.append("<option selected='selected' value='" + value + "'>" + value + "</option>");
 71 		}
 72 		selectTag.append("<option value='" + value + "'>" + value + "</option>");
 73 	});
 74 	if (defaultFound){
 75 		switch(option.lockDefault){
 76 			case 'disabled': 
 77 			selectTag.attr('disabled','disabled');
 78 			break;
 79 			case 'hidden':
 80 			createDiv.hide();
 81 			break;
 82 		}
 83 	}
 84 };

137 function rpcMilestone(name,title,append){
138 	try{
139 		var data=rpcTrac("ticket.milestone.getAll");
140 	}
141 	catch(err){
142 		throw err;
143 	}
144 	createDiv=$('<div>').addClass('tracSelect')
145 	.append("<label for='" + title + "'>" + title + "</label>").appendTo(append);
146 	selectTag=$('<select>').attr('id',name).attr('name',name).appendTo(createDiv);
147 	var defaultFound=false;
148 	$.each(data, function(index, value) {
149 		try{
150 			var result=rpcTrac("ticket.milestone.get",value);
151 		}
152 		catch(err){
153 			throw err;
154 		}
155 		//if the milestone is complete, it returns a date and not a number
156 		if(result.completed instanceof Number){
157 			if (option.defaultMilestone!= null && value.toLowerCase()==option.defaultMilestone.toLowerCase()){
158 				defaultFound=true;
159 				selectTag.append("<option selected='selected' value='" + value + "'>" + value + "</option>");
160 			}
161 			else{
162 				selectTag.append("<option value='" + value + "'>" + value + "</option>")
163 			}
164 		}
165 	});
166 	if (defaultFound){
167 		switch(option.lockDefault){
168 			case 'disabled': 
169 			selectTag.attr('disabled','disabled');
170 			break;
171 			case 'hidden':
172 			createDiv.hide();
173 			break;
174 		}
175 	}

186 function rpcTracTicketInfo(id,append){

189 		$(append).children().remove();
190 		$('<h2>').text('Ticket:' + data[0] + ' - Milestone:' + data[3].milestone).appendTo(append);

194 	catch(err){

199 /**
200 * Get available actions on a ticket and display them like Trac. Also remove previous actions
201 * @param id id of the ticket
202 * @param append where to append
203 */
204 
205 function rpcTracTicketAction(id,append){
206 	$(append).children().remove();
207 	try{
208 		var data=rpcTrac("ticket.getActions",id);
209 		$.each(data,function(index,value){
210 			var createDiv=$('<div>').addClass('tracAction').appendTo(append);
211 			$('<input>').attr('type','radio').attr('name','tracAction').attr('id',value[0]).attr('value',value[0]).appendTo(createDiv);
212 			$('<label>').attr('for',value[0]).text(value[1]).appendTo(createDiv);
213 			if( value[3].length > 0 && value[3][0][2].length > 0){
214 				var selectTag=$('<select>').attr('name',value[3][0][0]).attr('name',value[3][0][0]).appendTo(createDiv);
215 				$.each(value[3][0][2],function(index,value){
216 					selectTag.append("<option value='" + value + "'>" + value + "</option>")
217 				});
218 			}
219 			$('<div>').addClass('tracHint').text(value[2]).appendTo(createDiv);
220 		})
221 		$('input:radio[name=tracAction]').val([data[0][0]]);
222 	}
223 	catch(err){
224 		console.log("Most likely, unknow ticket. Error= " + err.faultString);
225 	}
226 
227 }

229 /**
230 * Test the connection to the trac.
231 * @returns {Boolean} True if connection is ok, else false
232 */
233 
234 function testTracConnection(){
235 	try{
236 		var data=rpcTrac("system.getAPIVersion");
237 	}
238 	catch(err){
239 		$('<div>').addClass('alert alert-error').attr('id','tracAlert').appendTo('#tracAppendZone');;
240 		if(err.code==101){
241 			$('<p>').text('101 error from XHR, check that you are connected and logged to the Trac server and that XSS are allowed').appendTo('#tracAlert');
242 		}
243 		else if(err.faultCode==403){
244 			$('<p>').text('403 error from Trac, check that you are connected and logged to the Trac server').appendTo('#tracAlert');
245 		}
246 		else{
247 			$('<p>').text('Unidentified error, check that you are connected and logged to the Trac server').appendTo('#tracAlert');
248 		}
249 		var retry=$('<button>').attr('data-dismiss','alert').attr('type','button').addClass('btn').text('Retry').appendTo('#tracAlert');
250 		retry.click(function(){
251 			$("#tracAlert").alert('close');
252 			createTracForm();
253 		});
254 		return false
255 	}
256 	return true
257 }

423 		var result = rpcTrac("ticket.putAttachment", id, "screenshot.png", "Automatic screenshot", base, false);
424 	}

433 * @param id id of the ticket
434 * @param timestamp timestamp of the ticket, it should match the _ts of the one provided by ticket info
435 */
436 
437 function updateTracTicket(id,timestamp){
438 	//document.write(ticket[3]._ts);
439 	var attributes={
440 		'_ts': timestamp,
441 		'action': $('input:radio[name=tracAction]:checked').val()
442 	}
443 	//retrieve value in select box if it exists
444 	var test = $('input:radio[name=tracAction]:checked').parent().children('select');
445 	if (test.length>0){
446 		attributes[test.attr('name')]=test.val();
447 	}
448 	try{
449 		return rpcTrac("ticket.update",id, $('#commentField').val()  + "\n" + $('#prefillTrac').html().replace(/<br>|<br\/>/g,'\n'),attributes);
450 	}
451 	catch(err){
452 		console.debug(err);
453 	}
454 }
